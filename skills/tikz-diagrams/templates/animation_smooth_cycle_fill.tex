\documentclass[border=5pt]{standalone}
\usepackage{tikz}
\usepackage{animate}
\usetikzlibrary{arrows.meta,calc}

% Reusable template for smooth-fill cycle animations.
% Use when the concept is continuous traversal, accumulation, phase movement,
% or progress around a loop. The path is drawn as completed quarter-arcs plus
% one short current arc; this avoids TikZ full-circle arc overflow near 360 deg.

\definecolor{accent}{RGB}{188,137,0}
\definecolor{ink}{RGB}{32,38,46}
\definecolor{grid}{RGB}{215,220,226}

\newcommand{\CycleBase}{%
  \path[use as bounding box] (-4.6,-4.0) rectangle (4.8,4.55);
  \node[font=\sffamily\bfseries\large, text=ink] at (0,4.2) {Smooth cycle fill};
  \draw[->, line width=0.65pt, ink] (-3.8,-3.25) -- (3.85,-3.25)
    node[below left=2pt, font=\scriptsize] {$x_t$};
  \draw[->, line width=0.65pt, ink] (-3.8,-3.25) -- (-3.8,3.45)
    node[below right=2pt, font=\scriptsize] {$y_t$};
  \draw[line width=0.55pt, grid] (-3.45,0) -- (3.45,0);
  \draw[line width=0.55pt, grid] (0,-3.0) -- (0,3.0);
  \draw[line width=0.85pt, ink] (0,0) circle[radius=2.55];
  \node[font=\scriptsize, align=center] at (-2.45,-2.75) {(1)\\State A};
  \node[font=\scriptsize, align=center] at (2.45,-2.75) {(2)\\State B};
  \node[font=\scriptsize, align=center] at (2.45,2.75) {(3)\\State C};
  \node[font=\scriptsize, align=center] at (-2.45,2.75) {(4)\\State D};
}

\newcommand{\SmoothCycleFrame}[1]{%
\begin{tikzpicture}[font=\sffamily, >=Stealth, ink/.style={draw=ink}]
  \CycleBase
  \ifnum#1<12
    \pgfmathsetmacro{\local}{#1}
    \pgfmathsetmacro{\ang}{-135+\local*90/11}
  \else\ifnum#1<24
    \draw[accent, line width=1.25pt, line cap=round]
      ({2.55*cos(-135)},{2.55*sin(-135)})
      arc[start angle=-135, end angle=-45, radius=2.55];
    \pgfmathsetmacro{\local}{#1-12}
    \pgfmathsetmacro{\ang}{-45+\local*90/11}
  \else\ifnum#1<36
    \draw[accent, line width=1.25pt, line cap=round]
      ({2.55*cos(-135)},{2.55*sin(-135)})
      arc[start angle=-135, end angle=-45, radius=2.55];
    \draw[accent, line width=1.25pt, line cap=round]
      ({2.55*cos(-45)},{2.55*sin(-45)})
      arc[start angle=-45, end angle=45, radius=2.55];
    \pgfmathsetmacro{\local}{#1-24}
    \pgfmathsetmacro{\ang}{45+\local*90/11}
  \else
    \draw[accent, line width=1.25pt, line cap=round]
      ({2.55*cos(-135)},{2.55*sin(-135)})
      arc[start angle=-135, end angle=-45, radius=2.55];
    \draw[accent, line width=1.25pt, line cap=round]
      ({2.55*cos(-45)},{2.55*sin(-45)})
      arc[start angle=-45, end angle=45, radius=2.55];
    \draw[accent, line width=1.25pt, line cap=round]
      ({2.55*cos(45)},{2.55*sin(45)})
      arc[start angle=45, end angle=135, radius=2.55];
    \pgfmathsetmacro{\local}{#1-36}
    \pgfmathsetmacro{\ang}{135+\local*90/11}
  \fi\fi\fi
  \coordinate (state) at ({2.55*cos(\ang)},{2.55*sin(\ang)});
  \draw[accent, line width=1.25pt, line cap=round, -{Stealth[length=4pt]}]
    ({2.55*cos(\ang-8)},{2.55*sin(\ang-8)})
    arc[start angle=\ang-8, end angle=\ang, radius=2.55];
  \fill[white] (state) circle[radius=5.5pt];
  \fill[accent] (state) circle[radius=3.2pt];
\end{tikzpicture}%
}

\begin{document}
\begin{animateinline}[poster=first,controls,loop]{8}
\multiframe{48}{i=0+1}{\SmoothCycleFrame{\i}}
\end{animateinline}
\end{document}
